Extending NoesisGUI

Being based in XAML, NoesisGUI is a framework that can be extended in many ways. For example, you can create your own UserControl or Commands. As we will discover soon, extending NoesisGUI in Unity is a lot easier than using C++.

Extending classes

Classes provided by NoesisGUI can be extended to provide new functionality. For example:

[Noesis.Extended]
public class Cell : Noesis.BaseComponent
{
    public string player
    {
        set { _player = value; }
        get { return _player; }
    }
    private string _player;

    public Noesis.ToggleButton button
    {
        set { _button = value; }
        get { return _button; }
    }
    private Noesis.ToggleButton _button;
};

Every extension always follows the same pattern:

  • The class attribute Noesis.Extended must be always used in the corresponding class.
  • You must always inherit from the a base class, BaseComponent in this case.

Example

The following example shows how to fill a ListBox using instances of an extended class. The first thing we do is implementing this class. Note how NotifyPropertyChanged is used to notify that one of our properties have changed.

[Noesis.Extended]
public class Player : Noesis.BaseComponent
{
    public Player(string name, Noesis.SolidColorBrush color, float scale, string pos)
    {
        _name = name;
        _color = color;
        _scale = scale;
        _pos = pos;
    }

    private string _name = string.Empty;
    public string Name
    {
        get { return _name; }
        set
        {
            if (_name != value)
            {
                _name = value;
                NotifyPropertyChanged("Name");
            }
        }
    }

    private Noesis.SolidColorBrush _color = null;
    public Noesis.SolidColorBrush Color
    {
        get { return _color; }
        set
        {
            if (_color != value)
            {
                _color = value;
                NotifyPropertyChanged("Color");
            }
        }
    }

    private float _scale = 1.0f;
    public float Scale
    {
        get { return _scale; }
        set
        {
            if (_scale != value)
            {
                _scale = value;
                NotifyPropertyChanged("Scale");
            }
        }
    }

    private string _pos = string.Empty;
    public string Pos
    {
        get { return _pos; }
        set
        {
            if (_pos != value)
            {
                _pos = value;
                NotifyPropertyChanged("Pos");
            }
        }
    }
}

Instances of this class can be inserted in a collection that serves as data source for the ListBox:

Grid root = GetComponent<NoesisGUIPanel>().GetRoot<Grid>();
ListBox listBox = root.FindName<ListBox>("ListBox");

Collection collection = new Collection();
collection.Add(new Player("Player0", new SolidColorBrush(Noesis.Color.Red), 1.0f, "(0,0,0)"));
collection.Add(new Player("Player1", new SolidColorBrush(Noesis.Color.Gray), 1.0f, "(0,30,0)"));
collection.Add(new Player("Player2", new SolidColorBrush(Noesis.Color.Orange), 1.0f, "(0,-10,0)"));
collection.Add(new Player("Player3", new SolidColorBrush(Noesis.Color.Green), 1.0f, "(0,-10,0)"));

listBox.SetItemsSource(collection);

And the XAML in charge or rendering this information could be as follow:

<Grid
    xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
    xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
    Width="600">

    <Grid.Resources>
        <DataTemplate x:Key="TaskTemplate">
            <Grid>
                <Grid.ColumnDefinitions>
                    <ColumnDefinition Width="Auto"/>
                    <ColumnDefinition Width="100"/>
                    <ColumnDefinition Width="200"/>
                    <ColumnDefinition Width="200"/>
                </Grid.ColumnDefinitions>

                <Rectangle Width="15" Height="10" Fill="{Binding Color}" Stroke="Transparent" StrokeThickness="0"
                    Margin="5, 0, 5, 0" Grid.Column="0"/>
                <TextBlock Text="{Binding Name}" Margin="5, 0, 5, 0" Grid.Column="1"/>
                <TextBlock Text="{Binding Scale, StringFormat=P0}" Margin="5, 0, 5, 0" Grid.Column="2"/>
                <TextBlock Text="{Binding Pos}" Margin="5, 0, 5, 0" Grid.Column="3"/>
            </Grid>
        </DataTemplate>
    </Grid.Resources>

    <ListBox x:Name="ListBox" Height="100" ItemTemplate="{StaticResource TaskTemplate}"></ListBox>
</Grid>
